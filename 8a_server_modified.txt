"""
Modulo para el servidor de deteccion de emociones.
"""
from flask import Flask, render_template, request
from EmotionDetection.emotion_detection import emotion_detector

app = Flask(__name__)

@app.route("/")
def render_index_page():
    """
    Renderiza la pagina principal.
    """
    return render_template('index.html')

@app.route("/emotionDetector")
def sent_analyzer():
    """
    Analiza emociones del texto recibido.
    """
    text_to_analyze = request.args.get('textToAnalyze')
    response = emotion_detector(text_to_analyze)

    if response['dominant_emotion'] is None:
        return "¡Texto no válido! ¡Por favor, inténtelo de nuevo!"

    traduccion = {
        'anger': 'ira',
        'disgust': 'asco',
        'fear': 'miedo',
        'joy': 'alegría',
        'sadness': 'tristeza'
    }
    emocion = traduccion.get(response['dominant_emotion'],
                             response['dominant_emotion'])

    return (f"Para la declaración dada, la respuesta del sistema es 'ira': "
            f"{response['anger']}, 'asco': {response['disgust']}, "
            f"'miedo': {response['fear']}, 'alegría': {response['joy']} y "
            f"'tristeza': {response['sadness']}. La emoción dominante es {emocion}.")

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
